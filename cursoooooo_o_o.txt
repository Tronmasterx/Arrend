DROP TABLE PAISES CASCADE CONSTRAINTS;
DROP TABLE MONEDAS CASCADE CONSTRAINTS;
DROP TABLE CURSOS CASCADE CONSTRAINTS;
DROP TABLE PROFESORES CASCADE CONSTRAINTS;
DROP TABLE ALUMNOS CASCADE CONSTRAINTS;
DROP TABLE HORARIOS CASCADE CONSTRAINTS;
DROP TABLE FACTURAS CASCADE CONSTRAINTS;

CREATE TABLE PAISES(
CLAVE_PAIS CHAR(4)  NOT NULL,
NOMBRE_PAIS VARCHAR2(20) NOT NULL,
CONSTRAINT PAIS_PK PRIMARY KEY(CLAVE_PAIS)
);

CREATE TABLE MONEDAS (
CLAVE_MONEDA CHAR(4) NOT NULL,
NOMBRE_MONEDA VARCHAR2(20) NOT NULL,
CONSTRAINT MONEDA_PK PRIMARY KEY(CLAVE_MONEDA)
);


CREATE TABLE CURSOS (
CLAVE_CURSO CHAR(4) NOT NULL,
NOMBRE_CURSO VARCHAR2(30) NOT NULL,
DESCRIPCION VARCHAR2(40) NOT NULL,
DURACION NUMBER (5) NOT NULL,
INVERSION NUMBER (5) NOT NULL,
CLAVE_MONEDA CHAR(4) NOT NULL,
BIBLIOGRAFIA VARCHAR2(40) ,
CONSTRAINT CURSOS_PK PRIMARY KEY(CLAVE_CURSO),
CONSTRAINT FK_CURSO_MONEDAS foreign key (CLAVE_MONEDA) references MONEDAS(CLAVE_MONEDA)
);

CREATE TABLE PROFESORES (
CLAVE_PROFESOR CHAR(4) NOT NULL,
CLAVE_PAIS CHAR(4) NOT NULL,
NOMBRE VARCHAR2(30) NOT NULL,
APELLIDO_PATERNO VARCHAR2(20) NOT NULL,
APELLIDO_MATERNO VARCHAR2(20) NOT NULL,
AREA_PROFESIONAL VARCHAR2(20) NOT NULL,
TEL_CONTACTO1 VARCHAR2(20) NOT NULL,
TEL_CONTACTO2 VARCHAR2(20),
GENERO VARCHAR2(10) NOT NULL,
CORREO_ELECTRONICO VARCHAR2(40) NOT NULL,
CONSTRAINT PROFESOR_PK PRIMARY KEY(CLAVE_PROFESOR),
CONSTRAINT FK_PROFESORES_PAIS foreign key (CLAVE_PAIS) references PAISES(CLAVE_PAIS)
);


CREATE TABLE ALUMNOS (
CLAVE_ALUMNO CHAR(4) NOT NULL,
CLAVE_PAIS CHAR(4) NOT NULL,
NOMBRE VARCHAR2(30) NOT NULL,
APELLIDO_PATERNO VARCHAR2(20) NOT NULL,
APELLIDO_MATERNO VARCHAR2(30) ,
AREA_PROFESIONAL VARCHAR2(20) NOT NULL,
TEL_CONTACTO1 VARCHAR2(20) NOT NULL,
TEL_CONTACTO2 VARCHAR2(20),
GENERO VARCHAR2(20) NOT NULL,
CORREO_ELECTRONICO VARCHAR2(40) NOT NULL,
CONSTRAINT AL_PK PRIMARY KEY(CLAVE_ALUMNO),
CONSTRAINT FK_ALUMNOS_PAIS foreign key (CLAVE_PAIS) references PAISES(CLAVE_PAIS)
);


CREATE TABLE HORARIOS(
CLAVE_HORARIO CHAR(4) NOT NULL,
CLAVE_PROFESOR CHAR(4) NOT NULL,
CLAVE_CURSO CHAR(4) NOT NULL,
FECHA_INICIAL DATE NOT NULL,
FECHA_FINAL DATE NOT NULL,
CONSTRAINT HORARIO_PK PRIMARY KEY(CLAVE_HORARIO),
CONSTRAINT FK_HORARIOS_PROFESORES foreign key (CLAVE_PROFESOR) references PROFESORES(CLAVE_PROFESOR),
CONSTRAINT FK_HORARIOS_CURSOS foreign key (CLAVE_CURSO) references CURSOS(CLAVE_CURSO)
);


CREATE TABLE FACTURAS(
CLAVE_FACTURA NUMBER NOT NULL,
CLAVE_CURSO CHAR(4) NOT NULL,
CLAVE_ALUMNO CHAR(4) NOT NULL,
FECHA_PAGO DATE NOT NULL,
CLAVE_MONEDA CHAR(4) NOT NULL,
INVERSION NUMBER(4) NOT NULL,
CONSTRAINT FACTURA_PK PRIMARY KEY(CLAVE_FACTURA),
CONSTRAINT FK_FACTURAS_ALUMNOS foreign key (CLAVE_ALUMNO) references ALUMNOS(CLAVE_ALUMNO),
CONSTRAINT FK_FACTURAS_CURSO foreign key (CLAVE_CURSO) references CURSOS(CLAVE_CURSO),
CONSTRAINT FK_FACTURAS_MONEDAS foreign key (CLAVE_MONEDA) references MONEDAS(CLAVE_MONEDA)
);
