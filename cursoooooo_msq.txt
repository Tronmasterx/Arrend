DROP TABLE IF EXISTS FACTURAS;
DROP TABLE IF EXISTS HORARIOS;
DROP TABLE IF EXISTS ALUMNOS;
DROP TABLE IF EXISTS PROFESORES;
DROP TABLE IF EXISTS CURSOS;
DROP TABLE IF EXISTS MONEDAS;
DROP TABLE IF EXISTS PAISES;

CREATE TABLE PAISES(
    CLAVE_PAIS CHAR(4) NOT NULL,
    NOMBRE_PAIS VARCHAR(20) NOT NULL,
    PRIMARY KEY (CLAVE_PAIS)
);

CREATE TABLE MONEDAS (
    CLAVE_MONEDA CHAR(4) NOT NULL,
    NOMBRE_MONEDA VARCHAR(20) NOT NULL,
    PRIMARY KEY (CLAVE_MONEDA)
);

CREATE TABLE CURSOS (
    CLAVE_CURSO CHAR(4) NOT NULL,
    NOMBRE_CURSO VARCHAR(30) NOT NULL,
    DESCRIPCION VARCHAR(40) NOT NULL,
    DURACION INT NOT NULL,
    INVERSION INT NOT NULL,
    CLAVE_MONEDA CHAR(4) NOT NULL,
    BIBLIOGRAFIA VARCHAR(40),
    PRIMARY KEY (CLAVE_CURSO),
    CONSTRAINT FK_CURSO_MONEDAS 
        FOREIGN KEY (CLAVE_MONEDA) REFERENCES MONEDAS(CLAVE_MONEDA)
);

CREATE TABLE PROFESORES (
    CLAVE_PROFESOR CHAR(4) NOT NULL,
    CLAVE_PAIS CHAR(4) NOT NULL,
    NOMBRE VARCHAR(30) NOT NULL,
    APELLIDO_PATERNO VARCHAR(20) NOT NULL,
    APELLIDO_MATERNO VARCHAR(20) NOT NULL,
    AREA_PROFESIONAL VARCHAR(20) NOT NULL,
    TEL_CONTACTO1 VARCHAR(20) NOT NULL,
    TEL_CONTACTO2 VARCHAR(20),
    GENERO VARCHAR(10) NOT NULL,
    CORREO_ELECTRONICO VARCHAR(40) NOT NULL,
    PRIMARY KEY (CLAVE_PROFESOR),
    CONSTRAINT FK_PROFESORES_PAIS 
        FOREIGN KEY (CLAVE_PAIS) REFERENCES PAISES(CLAVE_PAIS)
);

CREATE TABLE ALUMNOS (
    CLAVE_ALUMNO CHAR(4) NOT NULL,
    CLAVE_PAIS CHAR(4) NOT NULL,
    NOMBRE VARCHAR(30) NOT NULL,
    APELLIDO_PATERNO VARCHAR(20) NOT NULL,
    APELLIDO_MATERNO VARCHAR(30),
    AREA_PROFESIONAL VARCHAR(20) NOT NULL,
    TEL_CONTACTO1 VARCHAR(20) NOT NULL,
    TEL_CONTACTO2 VARCHAR(20),
    GENERO VARCHAR(20) NOT NULL,
    CORREO_ELECTRONICO VARCHAR(40) NOT NULL,
    PRIMARY KEY (CLAVE_ALUMNO),
    CONSTRAINT FK_ALUMNOS_PAIS 
        FOREIGN KEY (CLAVE_PAIS) REFERENCES PAISES(CLAVE_PAIS)
);

CREATE TABLE HORARIOS(
    CLAVE_HORARIO CHAR(4) NOT NULL,
    CLAVE_PROFESOR CHAR(4) NOT NULL,
    CLAVE_CURSO CHAR(4) NOT NULL,
    FECHA_INICIAL DATE NOT NULL,
    FECHA_FINAL DATE NOT NULL,
    PRIMARY KEY (CLAVE_HORARIO),
    CONSTRAINT FK_HORARIOS_PROFESORES 
        FOREIGN KEY (CLAVE_PROFESOR) REFERENCES PROFESORES(CLAVE_PROFESOR),
    CONSTRAINT FK_HORARIOS_CURSOS 
        FOREIGN KEY (CLAVE_CURSO) REFERENCES CURSOS(CLAVE_CURSO)
);

CREATE TABLE FACTURAS(
    CLAVE_FACTURA INT NOT NULL,
    CLAVE_CURSO CHAR(4) NOT NULL,
    CLAVE_ALUMNO CHAR(4) NOT NULL,
    FECHA_PAGO DATE NOT NULL,
    CLAVE_MONEDA CHAR(4) NOT NULL,
    INVERSION INT NOT NULL,
    PRIMARY KEY (CLAVE_FACTURA),
    CONSTRAINT FK_FACTURAS_ALUMNOS 
        FOREIGN KEY (CLAVE_ALUMNO) REFERENCES ALUMNOS(CLAVE_ALUMNO),
    CONSTRAINT FK_FACTURAS_CURSO 
        FOREIGN KEY (CLAVE_CURSO) REFERENCES CURSOS(CLAVE_CURSO),
    CONSTRAINT FK_FACTURAS_MONEDAS 
        FOREIGN KEY (CLAVE_MONEDA) REFERENCES MONEDAS(CLAVE_MONEDA)
);
