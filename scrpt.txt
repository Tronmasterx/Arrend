CREATE TABLE dbo.TB_BG_Cat_Categoria
	(
	Id_Categoria       CHAR (3) NOT NULL,
	Descripcion        VARCHAR (25) NULL,
	Num_Personas       CHAR (2) NULL,
	Num_Equipaje       CHAR (2) NULL,
	Num_Equipaje_Chico CHAR (2) NULL,
	CONSTRAINT PK_Categoria PRIMARY KEY (Id_Categoria)
	)
GO

CREATE TABLE dbo.TB_BG_Cat_Ciudad
	(
	Id_Ciudad CHAR (3) NOT NULL,
	Nombre    VARCHAR (25) NULL,
	CONSTRAINT PK_Ciudad PRIMARY KEY (Id_Ciudad)
	)
GO

CREATE TABLE dbo.TB_BG_Cat_Pais
	(
	Id_Codigo CHAR (3) NOT NULL,
	Pais      VARCHAR (25) NULL,
	CONSTRAINT PK_Pais PRIMARY KEY (Id_Codigo)
	)
GO

CREATE TABLE dbo.TB_BG_Cat_Sucursal
	(
	Id_Ciudad        CHAR (3) NOT NULL,
	Id_Sucursal      CHAR (3) NOT NULL,
	Nombre           VARCHAR (50) NULL,
	Direccion        VARCHAR (140) NULL,
	CP               CHAR (5) NULL,
	Telefono         VARCHAR (12) NULL,
	Horario_Apertura CHAR (5) NULL,
	Horario_Cierre   CHAR (5) NULL,
	CONSTRAINT PK_Sucursal PRIMARY KEY (Id_Ciudad,Id_Sucursal)
	)
GO

CREATE TABLE dbo.TB_BG_Cat_Tarifa
	(
	Id_Ciudad    CHAR (3) NOT NULL,
	Id_Categoria CHAR (3) NOT NULL,
	Costo_Dia    DECIMAL (8,2) NULL,
	CONSTRAINT PK_Tarifa PRIMARY KEY (Id_Ciudad,Id_Categoria)
	)
GO

CREATE TABLE dbo.TB_BG_Contrato
	(
	Id_Contrato       SMALLINT NOT NULL,
	Id_Fecha_Registro DATETIME NOT NULL,
	Id_Ciudad         CHAR (3) NOT NULL,
	Id_Sucursal       CHAR (3) NOT NULL,
	Id_Placas         CHAR (6) NULL,
	Fecha_Inicio      DATETIME NULL,
	Fecha_Fin         DATETIME NULL,
	Hora_Inicio       CHAR (5) NULL,
	Hora_Fin          CHAR (5) NULL,
	Costo_Dia         DECIMAL (8,2) NULL,
	Cond_Nombre       VARCHAR (30) NULL,
	Cond_Paterno      VARCHAR (25) NULL,
	Cond_Materno      VARCHAR (25) NULL,
	Cond_Edad         TINYINT NULL,
	Forma_Pago        VARCHAR (15) NULL,
	Tarjeta           VARCHAR (25) NULL,
	Num_Tarjeta       CHAR (16) NULL,
	Venc_Mes          CHAR (2) NULL,
	Venc_Anio         CHAR (4) NULL,
	Codigo_Seg        CHAR (4) NULL,
	Titular_Tarjeta   VARCHAR (50) NULL,
	Estatus           VARCHAR (9) NULL,
	CONSTRAINT PK_Contrato PRIMARY KEY (Id_Contrato,Id_Fecha_Registro)
	)
GO

CREATE TABLE dbo.TB_BG_Factura
	(
	Id_Contrato        SMALLINT NOT NULL,
	Id_Fecha_Registro  DATETIME NOT NULL,
	No_Factura         SMALLINT NULL,
	Tipo_Contribuyente CHAR (7) NULL,
	Nombre             VARCHAR (50) NULL,
	Calle              VARCHAR (35) NULL,
	Colonia            VARCHAR (35) NULL,
	CP                 CHAR (5) NULL,
	Municipio_Delega   VARCHAR (35) NULL,
	RFC                CHAR (13) NULL,
	Correo             VARCHAR (30) NULL,
	Telefono           VARCHAR (7) NULL,
	Id_Codigo          CHAR (4) NULL,
	Area               VARCHAR (6) NULL,
	Tel_Numero         VARCHAR (16) NULL,
	CONSTRAINT PK_Factura PRIMARY KEY (Id_Contrato,Id_Fecha_Registro)
	)
GO

CREATE TABLE dbo.TB_BG_Flotilla

	Id_Categoria       CHAR (3) NOT NULL,
